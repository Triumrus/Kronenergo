weather - 5 day / 3 hour forecast data (Прогноз погоды на 5 дней в разбивке по 3 часа)
SELECT `weather`.`date`,
    `weather`.`hour`,
    `weather`.`city_id`,
    `weather`.`temp`,
    `weather`.`temp_min`,
    `weather`.`temp_max`,
    `weather`.`pressure`,
    `weather`.`sea_level`,
    `weather`.`grnd_level`,
    `weather`.`humidity`,
    `weather`.`temp_kf`,
    `weather`.`weather_id`,
    `weather`.`main`,
    `weather`.`description`,
    `weather`.`icon`,
    `weather`.`alll`,
    `weather`.`speed`,
    `weather`.`deg`,
    `weather`.`rain`,
    `weather`.`snow`,
    `weather`.`pod`,
    `weather`.`id`
FROM `energo`.`weather`;
---
UNION historical_data_weather - weather
DATE(dt_iso_2) = date
HOUR(dt_iso_2) = hour
city_id = city_id
temp = temp
temp_min = temp_min
temp_max = temp_max
pressure = pressure
sea_level = sea_level
grnd_level = grnd_level
humidity = humidity
NA = temp_kf
weather_id = weather_id
weather_main = main
weather_description = description
weather_icon = icon
clouds_all = alll
wind_speed = speed
wind_deg = deg
